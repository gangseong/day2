#include <iostream>
long double probability(unsigned numbers, unsigned picks);
int main()
{
	using namespace std;
	double total, choices;
	long double first = 0.0;
	long double second = 0.0;

	cout << "전체 수의 개수와 뽑을 수의 개수를 입력하십시오:\n";
	cin >> total >> choices;
	first = probability(total, choices);
	cout << "다시 두 수를 입력하십시오(메가 볼). \n";
	cin >> total >> choices;
	second=probability(total, choices);
	cout << "당신이 이길 확률은 ";
	cout << first * second;
	cout << "번 중에서 한 번입니다.\n";
	
	return 0;
}
long double probability(unsigned numbers, unsigned picks)
{
	long double result = 1.0;
	long double n;
	unsigned p;

	for (n = numbers, p = picks; p > 0; n--, p--)
		result = result * n / p;
	return result;
}
